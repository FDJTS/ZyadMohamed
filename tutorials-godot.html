المتطلبات

  • Godot Engine 4.x
  • Visual Studio Code أو أي محرر نصوص
  • معرفة أساسية بـ C++
  • معرفة أساسية بـ GDScript

خطوات إنشاء المشروع

1. إنشاء مشروع جديد في Godot
# افتح Godot Engine
# اختر "New Project"
# اختر مجلد للمشروع
# اختر "Create & Edit"
2. إعداد البيئة للـ C++
# في Godot، اذهب إلى:
# Project → Tools → C# → Create C# solution
# أو استخدم GDExtension للـ C++
3. إنشاء المشهد الرئيسي
# أضف Node3D كـ root
# أضف Camera3D كـ child
# أضف MeshInstance3D للكائنات

كود C++ أساسي

إنشاء كلاس Player
#include <godot_cpp/classes/character_body3d.hpp>
#include <godot_cpp/classes/input.hpp>

using namespace godot;

class Player : public CharacterBody3D {
    GDCLASS(Player, CharacterBody3D)
    
private:
    float speed = 5.0f;
    float jump_velocity = 4.5f;
    
public:
    void _ready() override {
        // تهيئة اللاعب
    }
    
    void _physics_process(double delta) override {
        Vector3 velocity = get_velocity();
        
        // الجاذبية
        if (!is_on_floor()) {
            velocity.y -= get_gravity().y * delta;
        }
        
        // القفز
        if (Input::get_singleton()->is_action_just_pressed("ui_accept") && is_on_floor()) {
            velocity.y = jump_velocity;
        }
        
        // الحركة
        Vector2 input_dir = Input::get_singleton()->get_vector("ui_left", "ui_right", "ui_up", "ui_down");
        Vector3 direction = Vector3(input_dir.x, 0, input_dir.y);
        
        if (direction != Vector3()) {
            velocity.x = direction.x * speed;
            velocity.z = direction.z * speed;
        } else {
            velocity.x = Math::move_toward(velocity.x, 0, speed);
            velocity.z = Math::move_toward(velocity.z, 0, speed);
        }
        
        set_velocity(velocity);
        move_and_slide();
    }
};

REGISTER_GODOT_CLASS(Player)

إعداد التحكم

في Project Settings → Input Map، أضف:
  • ui_left: A أو Left Arrow
  • ui_right: D أو Right Arrow
  • ui_up: W أو Up Arrow
  • ui_down: S أو Down Arrow
  • ui_accept: Space أو Enter
